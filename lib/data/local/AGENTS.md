### DAO Pattern
```dart
@DriftAccessor(tables: [RecipesTable, RecipeStepsTable, IngredientsTable])
class RecipeDao extends DatabaseAccessor<AppDatabase> {
  @override
  final AppDatabase db;
  
  @Query('SELECT * FROM recipes WHERE id = :id')
  Future<Recipe?> getRecipeById(int id);
  
  Stream<List<Recipe>> watchRecipesByCategory(String category) {
    return (select(recipes)..where((r) => r.category.equals(category))).watch();
  }
}
```
